drwxr-x--- ... root root ... /root
# fam

% groups | grep root

% ./test -d /root
FAMMonitorDirectory("/root")

DIR /root: /root Exists
DIR /root: .gnome Exists
DIR /root: Desktop Exists
...

CORRECT BEHAVIOR
% ./test -d /root
FAMMonitorDirectory("/root")
FAMMonitorDirectory("/root")
DIR /root: /root Exists
DIR /root: /root EndExist
